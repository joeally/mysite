def find_angle(v):
    if np.linalg.norm(v) == 0:
        return 0
    xaxis = [1, 0]
    cos_sim = np.dot(xaxis, v)/np.linalg.norm(v)
    return np.arccos(cos_sim)

#Obtain the angles at which the car is travelling for each of the trips
thetas = [np.apply_along_axis(find_angle, 1, deltas) for deltas in data_dt_mov]
